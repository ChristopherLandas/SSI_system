CREATE DATABASE ssi;

CREATE TABLE acc_cred (usn VARCHAR(128) PRIMARY KEY COLLATE LATIN1_GENERAL_CS, pss VARCHAR(128) NOT NULL COLLATE LATIN1_GENERAL_CI, slt VARCHAR (64) NOT NULL COLLATE LATIN1_GENERAL_CI, entry_OTP varchar(128) COLLATE LATIN1_GENERAL_CI );
CREATE TABLE acc_info (usn VARCHAR(128) PRIMARY KEY COLLATE LATIN1_GENERAL_CS, full_name VARCHAR(128) NOT NULL COLLATE LATIN1_GENERAL_CI, job_position VARCHAR(32) NOT NULL COLLATE LATIN1_GENERAL_CI, FOREIGN KEY(usn) REFERENCES acc_cred(usn));
CREATE TABLE log_history (usn VARCHAR(128) NOT NULL COLLATE LATIN1_GENERAL_CS, date_logged DATE NOT NULL, time_in TIME NOT NULL, time_out TIME NOT NULL, FOREIGN KEY (usn) REFERENCES acc_cred(usn));
CREATE TABLE action_history (usn VARCHAR(12) PRIMARY KEY NOT NULLCOLLATE LATIN1_GENERAL_CS, ACTION VARCHAR (256) NOT NULL COLLATE LATIN1_GENERAL_CI, action_date DATE NOT NULL, FOREIGN KEY (usn) REFERENCES acc_cred(usn));

CREATE TABLE Item_General_Info (UID VARCHAR(12) PRIMARY KEY NOT NULL COLLATE LATIN1_GENERAL_CI, name VARCHAR(64) NOT NULL COLLATE LATIN1_GENERAL_CI, Category VARCHAR(64) NOT NULL COLLATE LATIN1_GENERAL_CI, Manufacturer VARCHAR(64) NOT NULL COLLATE LATIN1_GENERAL_CI);
CREATE TABLE Item_Inventory_Info (UID VARCHAR(12) NOT NULL COLLATE LATIN1_GENERAL_CI, Stock INT NOT NULL, Expiry_Date DATE, FOREIGN KEY (UID) REFERENCES Item_General_Info(UID));
CREATE TABLE Item_Supplier_Info (UID VARCHAR(12) PRIMARY KEY NOT NULL COLLATE LATIN1_GENERAL_CI, Supplier VARCHAR(64) NOT NULL COLLATE LATIN1_GENERAL_CI, Contacts VARCHAR(64) NOT NULL COLLATE LATIN1_GENERAL_CI, FOREIGN KEY (UID) REFERENCES Item_General_Info(UID));
CREATE TABLE Item_Settings (UID VARCHAR(12) PRIMARY KEY NOT NULL COLLATE LATIN1_GENERAL_CI, Price FLOAT NOT NULL, Reorder_factor FLOAT NOT NULL, Crit_factor FLOAT NOT NULL, Safe_stock INT NOT NULL, FOREIGN KEY (UID) REFERENCES Item_General_Info(UID));
